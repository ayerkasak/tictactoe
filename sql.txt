// Basic INNER JOIN Syntax

SELECT column1, column2
FROM table1
INNER JOIN table2
ON table1.common_column = table2.common_column;

//Example 1:
SELECT members.name, payments.amount
FROM members
INNER JOIN payments
ON members.member_id = payments.member_id;
