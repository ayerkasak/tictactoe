//Creating a table in database
 CREATE TABLE members (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT,
    age INTEGER,
    membership TEXT
 );

// Using WHERE for filtering data

SELECT * 
FROM members

SELECT * 
FROM members

FROM members
WHERE age = 20;
